                    const response = await fetch("https://formsubmit.co/ajax/biuro@damex24.pl", {
                        method: "POST",
                        body: data

                        setIsSuccess(true);
                        setFormData({
                            name: '',
                            company: '',
                            phone: '',
                            type: 'Biuro / Instytucja',

                        alert("Wystąpił problem z wysłaniem wiadomości. Prosimy o kontakt telefoniczny.");

                    console.error("Błąd:", error);
